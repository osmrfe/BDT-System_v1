    const animateCounter = (counter) => {

        let currentCount = 0;
        counter.innerText = currentCount;

        const updateCount = () => {
            const increment = target / counterSpeed;
            currentCount += increment;

                const counterElement = entry.target.querySelector('.counter') || entry.target;
